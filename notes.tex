\begin{enumerate}
  \item Install Git:
        \begin{enumerate}
          \item On Mac OS / Linux:
                \begin{enumerate}
                  \item Open Terminal
                  \item Type \code{apt-get install git}
                \end{enumerate}
          \item On Windows: go to the Git \href{https://git-scm.com/install/windows}{website}, download and follow the installation instructions.
        \end{enumerate}
  \item Install Paraview:
        On MacOS / Linux / Windows:
        Go to the Paraview \href{https://www.paraview.org/download/}{website},  download the latest version for your operating system, and follow the installation instructions.
  \item Install Docker desktop:
        \begin{enumerate}
          \item   On MacOS / Linux / Windows: go to the Docker \href{https://www.docker.com/products/docker-desktop/}{website}, download it and follow the installation instructions.
          \item Start Docker desktop.
        \end{enumerate}

  \item Download \libname{}:
        \begin{enumerate}
          \item On Mac OS:
                \begin{enumerate}
                  \item Open Terminal
                  \item \code{mkdir /Users/michele/Documents/irene} (change the path name according to your needs)
                  \item \code{cd /Users/michele/Documents/irene}
                  \item
                        \begin{minted}{bash}
git clone https:\\github.com\Michele-s-team\finite_elements.git
  chmod -R 777 finite_elements
\end{minted}
                \end{enumerate}
          \item On Windows:
                \begin{enumerate}
                  \item Open Terminal
                  \item \code{mkdir C:\textbackslash{}Users\textbackslash{}michele\textbackslash{}Documents\textbackslash{}irene}
                  \item \code{cd C:\textbackslash{}Users\textbackslash{}michele\textbackslash{}Documents\textbackslash{}irene}
                  \item
                        \begin{minted}{bash}
git clone https://github.com/Michele-s-team/finite_elements.git
                  \end{minted}
                \end{enumerate}
        \end{enumerate}

  \item Create the virtual environment:
        \begin{enumerate}
          \item On Mac OS / Linux:
                \begin{enumerate}
                  \item Open Terminal
                  \item \code{cd /Users/michele/Documents/irene}
                  \item \code{sudo docker pull michelecastellana/fenics\_image}
                \end{enumerate}

          \item On Windows
                \begin{enumerate}
                  \item \code{cd C:\textbackslash{}Users\textbackslash{}michele\textbackslash{}Documents\textbackslash{}irene}
                  \item \code{docker pull michelecastellana/fenics\_image}
                \end{enumerate}
        \end{enumerate}
\end{enumerate}



You can then start the virtual environment with
\begin{itemize}
  \item On Mac OS / Linux:\\


        \begin{minted}{bash}
docker run -ti \
          -v $(pwd)/finite_elements:/home/fenics/shared \
          --name fe_1 fenics_image
\end{minted}

  \item On Windows:
        \begin{minted}{bash}
docker run -ti \
        -v "C:\Users\michele\Documents\irene":/home/fenics/shared \
        michelecastellana/fenics_image:latest
\end{minted}


  \item Make a test run:
        \begin{itemize}
          \item Generate the mesh:
                \begin{minted}{bash}
cd /home/fenics/shared/generate_mesh/2d/square

                \end{minted}
        \end{itemize}
\end{itemize}
